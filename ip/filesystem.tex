\section{Introduction}

A file system has named files so we introduce file name and file content types.
\begin{zed}
[Name, File] 
\end{zed}

\begin{zed}
 Path == \seq Name
\end{zed}

\begin{schema}{FS}
 fs : Path \pfun File
\end{schema}

\begin{schema}{CreateFile}
 \Delta FS \\
 p? : Path \\
 f? : File \\
\where
 p? \notin \dom fs \\
 fs' = fs \cup \{ p? \mapsto f? \}
\end{schema}

\begin{schema}{DeleteFile}
 \Delta FS \\
 p? : Path \\
\where
 p? \in \dom fs \\
 fs' = \{ p? \} \ndres fs
\end{schema}
